import { SelectionGroup, Checkbox, RadioButton } from 'hds-react';
import TabsLayout from './tabs.mdx';

export default ({ children, pageContext }) => <TabsLayout pageContext={pageContext}>{children}</TabsLayout>;

## Usage

### Example

export const SelectionGroupExample = () => {
  const [checkedItems, setCheckedItems] = React.useState({});
  const onChange = (event) => {
    setCheckedItems({ ...checkedItems, [event.target.name]: event.target.checked });
  };
  return (
    <SelectionGroup label="Selection group" helperText="Assistive text">
      <Checkbox id="v-checkbox1" name="v-checkbox1" label="Option" checked={checkedItems["v-checkbox1"]} onChange={onChange} />
      <Checkbox id="v-checkbox2" name="v-checkbox2" label="Option" checked={checkedItems["v-checkbox2"]} onChange={onChange} />
      <Checkbox id="v-checkbox3" name="v-checkbox3" label="Option" checked={checkedItems["v-checkbox3"]} onChange={onChange} />
      <Checkbox id="v-checkbox4" name="v-checkbox4" label="Option" checked={checkedItems["v-checkbox4"]} onChange={onChange} />
    </SelectionGroup>
  );
};

### Principles
- Currently, Selection group can contain either Checkboxes or Radio buttons.
  - If the user can select more than one option from a list, use <InternalLink href="/components/checkbox">checkboxes.</InternalLink>
  - If the user can only select one option from a list, use <InternalLink href="/components/radio-button">radio buttons.</InternalLink>
- It is not recommended to have a default option selected because
  - The user might not realise theyâ€™ve missed a question
  - The user might submit a wrong answer
- Visible label must be given to the Selection group. The label should be clear and describe the choice the user is about to make.
- Selection group can be aligned both vertically and horizontally. **Vertical alignment is recommended.** Horizontal alignment can be used when there are less than 4 options.
  -  Note that horizontal Selection groups must not span to multiple lines.

### Variations

#### Vertical group
Vertical selection group is the recommended way to present grouped form controls. Vertical checkbox groups are generally easier to perceive and read.

export const VerticalGroupExample = () => {
  const [checkedItems, setCheckedItems] = React.useState({});
  const onChange = (event) => {
    setCheckedItems({ ...checkedItems, [event.target.name]: event.target.checked });
  };
  return (
    <SelectionGroup label="Vertical group">
      <Checkbox id="v-checkbox5" name="v-checkbox5" label="Option" checked={checkedItems["v-checkbox5"]} onChange={onChange} />
      <Checkbox id="v-checkbox6" name="v-checkbox6" label="Option" checked={checkedItems["v-checkbox6"]} onChange={onChange} />
      <Checkbox id="v-checkbox7" name="v-checkbox7" label="Option" checked={checkedItems["v-checkbox7"]} onChange={onChange} />
      <Checkbox id="v-checkbox8" name="v-checkbox8" label="Option" checked={checkedItems["v-checkbox8"]} onChange={onChange} />
    </SelectionGroup>
  );
};

#### Horizontal group
Horizontal selection group can be used in situations where there are only a few (less than 4) options. Otherwise, a vertical alignment is recommended.

export const HorizontalGroupExample = () => {
  const [selectedItem, setSelectedItem] = React.useState();
  const onChange = (event) => {
    setSelectedItem(event.target.value);
  };
  return (
    <SelectionGroup label="Horizontal group" direction="horizontal">
      <RadioButton id="h-radio1" name="h-radio" value="1" label="Option" checked={selectedItem === "1"} onChange={onChange} />
      <RadioButton id="h-radio2" name="h-radio" value="2" label="Option" checked={selectedItem === "2"} onChange={onChange} />
      <RadioButton id="h-radio3" name="h-radio" value="3" label="Option" checked={selectedItem === "3"} onChange={onChange} />
    </SelectionGroup>
  );
};
