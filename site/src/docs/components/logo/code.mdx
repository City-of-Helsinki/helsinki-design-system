### Code examples

#### Default and colour variants

<Playground scope={{ Logo, LogoSize, logoFi, logoFiDark, logoSv, logoSvDark, logoRu, logoRuDark }}>

```jsx
import { Logo, LogoSize, logoFi, logoFiDark, logoSv, logoSvDark, logoRu, logoRuDark } from 'hds-react';

() => (
  <>
    <Logo src={logoFi} size={LogoSize.Medium} style={{ marginRight: 'var(--spacing-xs)' }} alt="Helsingin kaupunki" />
    <Logo src={logoSv} size={LogoSize.Medium} style={{ marginRight: 'var(--spacing-xs)' }} alt="Helsingin kaupunki" />
    <Logo src={logoRu} size={LogoSize.Medium} style={{ marginRight: 'var(--spacing-xs)' }} alt="Helsingin kaupunki" />
    <Logo
      src={logoFiDark}
      size={LogoSize.Medium}
      style={{ marginRight: 'var(--spacing-xs)', background: 'black' }}
      alt="Helsingin kaupunki"
    />
    <Logo
      src={logoSvDark}
      size={LogoSize.Medium}
      style={{ marginRight: 'var(--spacing-xs)', background: 'black' }}
      alt="Helsingin kaupunki"
    />
    <Logo
      src={logoRuDark}
      size={LogoSize.Medium}
      style={{ marginRight: 'var(--spacing-xs)', background: 'black' }}
      alt="Helsingin kaupunki"
    />
  </>
);
```

</Playground>
