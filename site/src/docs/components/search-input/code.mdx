### Code examples 

#### Default

<PlaygroundBlock>

```html
<div style="max-width: 320px;">
  <div class="hds-search-input" id="search-example-container">
  <label for="search-example" class="hds-search-input__label">Search</label>
  <div class="hds-search-input__input-wrapper">
    <input
      id="search-example"
      class="hds-search-input__input"
      type="text"
      placeholder="Placeholder text"
      oninput="(function() {
        const input = document.getElementById('search-example');
        const clearButton = document.getElementById('search-example-clear');
        if (input && clearButton && input.value && input.value.length > 0) {
          clearButton.style.display = '';
        } else if (clearButton) {
          clearButton.style.display = 'none';
        }
      })()"
      onkeydown="if (event.key === 'Enter') { event.preventDefault(); const btn = document.getElementById('search-example-search'); if (btn) btn.click(); }"
    />
    <div class="hds-search-input__buttons">
      <button
        type="button"
        id="search-example-clear"
        aria-label="Clear search field"
        class="hds-search-input__button"
        style="display: none;"
        onclick="(function() {
          const input = document.getElementById('search-example');
          const clearButton = document.getElementById('search-example-clear');
          if (input) {
            input.value = '';
            input.focus();
          }
          if (clearButton) {
            clearButton.style.display = 'none';
          }
        })()"
      >
        <i class="hds-icon hds-icon--cross-circle hds-icon--size-s" aria-hidden="true"></i>
      </button>
      <button
        type="button"
        id="search-example-search"
        aria-label="Search"
        class="hds-search-input__button"
        onclick="(function() {
          const input = document.getElementById('search-example');
          if (input && input.value !== null && input.value !== undefined) {
            alert('Search for: ' + input.value);
          }
        })()"
      >
        <i class="hds-icon hds-icon--search hds-icon--size-s" aria-hidden="true"></i>
      </button>
    </div>
  </div>
  <span class="hds-search-input__helper-text">Assistive text</span>
  </div>
</div>
```
