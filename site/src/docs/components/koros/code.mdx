  const rootStyle = { '--koros-height': '85px', '--hero-height': '300px', '--hero-width': '100%' };

    <div
      style={{
        ...rootStyle,

        position: 'relative',
        width: 'var(--hero-width)',

      />
      <Koros
        style={{
          fill: 'var(--color-coat-of-arms)',
          left: 'calc(-1 * var(--koros-height))',
          position: 'absolute',

          width: 'calc(2 * var(--hero-height))',
        }}
        rotate="135deg"
      />
    </div>
  );
}
