### Code examples

#### Default

<Playground scope={{ StepByStep }}>

```jsx
import { StepByStep } from 'hds-react';

() => {
  const steps = [
    {
      title: 'Step 1 title',
      description:
        "Here you can describe the step in detail. Keep the text compact so the user gets the big picture of the whole process and it's steps easily.",
      buttons: [
        {
          children: 'Example button',
          href: 'https://hel.fi',
          key: 'button',
        },
      ],
    },
    {
      title: 'Step 2 title',
      description: 'Step description.',
      links: [
        {
          children: 'Example link',
          href: 'https://hel.fi',
          key: 'link',
        },
      ],
    },
    {
      title: 'Step 3 title',
      description: 'Step description.',
    },
  ];

  return (
    <StepByStep
      title="Unnumbered step by step component"
      helpText="Unnumbered component is suitable for cases where the order of the steps is less important and they form more of a guideline."
      steps={steps}
    />
  );
}
```
