## Customisation

### Customisation properties

You can use the `theme` property to customise the component. See all available theme variables in the table below.

| Theme variable                                      | Description                                                              |

| `--search-background-default`                       | Background color of the search input.                                    |

### Customisation example

<Playground scope={{ Search, Notification }}>

```jsx
import { Search, Notification } from 'hds-react';

() => {
  const [showNotification, setShowNotification] = React.useState(false);
  const [notificationMessage, setNotificationMessage] = React.useState('');

  const fruits = [
    'Apple', 'Apricot', 'Avocado', 'Banana', 'Blackberry', 'Blueberry', 'Cherry', 'Coconut',
    'Cranberry', 'Dragon fruit', 'Fig', 'Grape', 'Grapefruit', 'Guava', 'Honeydew', 'Kiwi',
    'Lemon', 'Lime', 'Lychee', 'Mango', 'Nectarine', 'Orange', 'Papaya', 'Passion fruit',
    'Peach', 'Pear', 'Pineapple', 'Plum', 'Pomegranate', 'Raspberry', 'Starfruit',
    'Strawberry', 'Tangerine', 'Watermelon',
  ];

  const handleSearch = React.useCallback((searchValue) => {
    return Promise.resolve({
      options: fruits
        .filter((s) => s.toLowerCase().indexOf(searchValue.toLowerCase()) >= 0)
        .map((s) => ({ label: s, value: s })),
    });
  }, []);

  const [props] = React.useState({
    id: 'search-themed',
    texts: {
      label: 'Themed search',
      language: 'en',
    },
    theme: {
      '--search-background-default': '#1a0a2e',
      '--search-border-color-default': '#ff6ec7',
      '--search-border-color-hover': '#ff99d6',
      '--search-border-color-focus': '#00ffcc',
      '--search-color-default': '#ffffff',
      '--search-icon-color': '#ff6ec7',
      '--placeholder-color': '#b088cc',
      '--text-label-color-default': '#ff6ec7',
      '--text-input-background-default': '#1a0a2e',
      '--text-input-border-color-default': '#ff6ec7',
      '--text-input-border-color-hover': '#ff99d6',
      '--text-input-border-color-focus': '#00ffcc',
      '--text-input-color-default': '#ffffff',
      '--text-input-placeholder-color': '#b088cc',
      '--menu-item-background-color-default': '#2d1b4e',
      '--menu-item-background-color-hover': '#ff6ec7',
      '--menu-item-color-default': '#ffffff',
      '--menu-item-color-hover': '#1a0a2e',
      '--clear-button-color': '#ff6ec7',
      '--clear-button-color-hover': '#00ffcc',
      '--clear-button-background-color-hover': '#2d1b4e',
      '--focus-outline-color': '#00ffcc',
    },
  });

  return (
    <>
      {showNotification && (
        <Notification
          label="Search submitted"
          type="success"
          position="bottom-right"
          autoClose
          closeButtonLabelText="Close notification"
          dismissible
          onClose={() => setShowNotification(false)}
        >
          {notificationMessage}
        </Notification>
      )}
      <Search
        {...props}
        onSearch={handleSearch}
        onSend={(val) => {
          setNotificationMessage(`Search submitted: "${val}"`);
          setShowNotification(true);
        }}
      />
    </>
  );
};
```
