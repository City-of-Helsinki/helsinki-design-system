## Code

### Code examples 

#### Default
<Playground scope={{ ToggleButton }}>

```jsx
import { ToggleButton } from 'hds-react';

() => {
  const [checked1, setChecked1] = React.useState(false);
  const [checked2, setChecked2] = React.useState(false);
  return (<>
    <ToggleButton id="toggle-button-1" label="Label" checked={checked1} onChange={() => setChecked1(!checked1)} />
    <br />
    <ToggleButton
      id="toggle-button-default"
      disabled
      label="Label"
      checked={checked2}
      onChange={() => setChecked2(!checked2)}
    />
  </>);
}
```

</Playground>

#### With a horizontal label
<Playground scope={{ ToggleButton }}>

```jsx
import { ToggleButton } from 'hds-react';

() => {
  const [checked1, setChecked1] = React.useState(false);
  const [checked2, setChecked2] = React.useState(false);
  return (<>
    <div style={{ maxWidth: '160px' }}>
      <ToggleButton
        id="toggle-button-1"
        label="Label"
        checked={checked1}
        onChange={() => setChecked1(!checked1)}
        variant="inline"
      />
      <br />
      <ToggleButton
        id="toggle-button-2"
        disabled
        label="Label"
        checked={checked2}
        onChange={() => setChecked2(!checked2)}
        variant="inline"
      />
    </div>
  </>);
}
```
