export const TabsElement = ({ initiallyActiveTab, children }) => {
  const config = [
    {
      path: 'usage',
      label: 'Usage',
    },
    {
      path: 'code',
      label: 'Code',
    },
    {
      path: 'accessibility',
      label: 'Accessibility',
    },
    {
      path: 'customisation',
      label: 'Customisation',
    },
  ];
  const activeIndex = config.findIndex(({ path }) => path === initiallyActiveTab);
  return (
    <Tabs initiallyActiveTab={activeIndex}>
      <Tabs.TabList className="pageTabsList">
        {config.map(({ path, label }) => (
          <Tabs.Tab onClick={() => navigate(`/elements/components/buttons/${path}`)}>{label}</Tabs.Tab>
        ))}
      </Tabs.TabList>
      {config.map(({ path, label }, index) => (
        <Tabs.TabPanel>{index === activeIndex ? children : label}</Tabs.TabPanel>
      ))}
    </Tabs>
  );
};

<TabsElement initiallyActiveTab={props.initiallyActiveTab}>{props.children}</TabsElement>

export default ({ children }) => <>{children}</>;
