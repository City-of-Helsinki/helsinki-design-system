#### Loading button

<Playground>


  ```jsx
  {
    () => {
      const [isLoading, setIsLoading] = React.useState(false);
      const [showNotification, setShowNotification] = React.useState(false);
      React.useEffect(() => {
        let timeout;
        if (isLoading) {
          timeout = setTimeout(() => {
            setShowNotification(true);
            setIsLoading(false);
          }, 2000);
        }
        return () => {
          clearTimeout(timeout);
        };
      }, [isLoading]);
      return (
        <>
          <Button
            isLoading={isLoading}
            loadingText="Saving form changes"
            onClick={() => {
              setShowNotification(false);
              setIsLoading(true);
            }}
          >
            Save form
          </Button>
          {showNotification && (
            <Notification
              key={new Date().toString()}
              position="top-right"
              displayAutoCloseProgress={false}
              autoClose
              dismissible
              label="Form saved!"
              type="success"
              onClose={() => {
                setShowNotification(false);
              }}
            >
              Saving your form was successful.
            </Notification>
          )}
        </>
      );
    }
  }
  ```
